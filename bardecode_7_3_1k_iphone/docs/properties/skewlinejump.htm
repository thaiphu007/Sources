MaxRectOverlap

Overview

The maximum percentage overlap allowed between 2 bounding rectangles.

Default value =  70

void STSetParameter(hBarcode, ST_MAX_RECT_OVERLAP, uint16 *value) ;

void STGetParameter(hBarcode,  ST_MAX_RECT_OVERLAP, uint16 *value) ;

e.g

#include "barcode.h"
uint16 value = 1 ;
STSetParameter(hBarcode, ST_MAX_RECT_OVERLAP, &value) ;
